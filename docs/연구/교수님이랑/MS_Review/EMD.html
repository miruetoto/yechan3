import numpy as np
import matplotlib.pyplot as plt

# 샘플링 주파수 및 시간 축 설정
fs = 1000
t = np.linspace(0, 2, 2 * fs, endpoint=False)

# 두 개의 주파수를 가진 신호 생성
signal = np.sin(2 * np.pi * 5 * t) + 0.5 * np.sin(2 * np.pi * 25 * t)

plt.figure(figsize=(10, 4))
plt.plot(t, signal)
plt.title("Composite Signal with Two Frequencies")
plt.xlabel("Time [s]")
plt.ylabel("Amplitude")
plt.show()
import numpy as np
import matplotlib.pyplot as plt
from PyEMD import EMD

# EMD 수행
emd = EMD()
IMFs = emd.emd(signal)
nIMFs = IMFs.shape[0]

plt.figure(figsize=(10, 8))
plt.subplot(nIMFs+1, 1, 1)
plt.plot(t, signal)
plt.title("Original Signal")
for n in range(nIMFs):
    plt.subplot(nIMFs+1, 1, n+2)
    plt.plot(t, IMFs[n])
    plt.title(f"IMF {n+1}")
plt.tight_layout()
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# 샘플링 주파수 및 시간 축 설정
fs = 1000
t = np.linspace(0, 2, 2 * fs, endpoint=False)

# 두 개의 주파수를 가진 신호 생성
signal = np.sin(2 * np.pi * 5 * t) + 0.5 * np.sin(2 * np.pi * 25 * t)

plt.figure(figsize=(10, 4))
plt.plot(t, signal)
plt.title("Composite Signal with Two Frequencies")
plt.xlabel("Time [s]")
plt.ylabel("Amplitude")
plt.show()
import numpy as np
import matplotlib.pyplot as plt
from scipy.interpolate import CubicSpline

def sift_once(signal, t):
    # 최대값과 최소값을 찾기
    max_peaks = [j for j in range(1, len(signal)-1) if signal[j-1] < signal[j] > signal[j+1]]
    min_peaks = [j for j in range(1, len(signal)-1) if signal[j-1] > signal[j] < signal[j+1]]

    # 최대값과 최소값의 위치에 대해 cubic spline 보간
    if len(max_peaks) > 1 and len(min_peaks) > 1:
        upper_env = CubicSpline(t[max_peaks], signal[max_peaks])(t)
        lower_env = CubicSpline(t[min_peaks], signal[min_peaks])(t)
        
        # 평균 곡선 계산
        mean_env = (upper_env + lower_env) / 2
        
        # IMF 갱신
        imf = signal - mean_env
        
        return imf, upper_env, lower_env, mean_env
    else:
        return signal, np.zeros_like(signal), np.zeros_like(signal), np.zeros_like(signal)

# 초기 신호 설정
residual = signal
imfs = []

# 체거름 과정 시각화
for i in range(5):
    imf, upper_env, lower_env, mean_env = sift_once(residual, t)
    imfs.append(imf)
    
    plt.figure(figsize=(10, 6))
    plt.subplot(2, 1, 1)
    plt.plot(t, residual, 'b', label='Residual')
    plt.plot(t, upper_env, 'r', label='Upper Envelope')
    plt.plot(t, lower_env, 'g', label='Lower Envelope')
    plt.plot(t, mean_env, 'orange', label='Mean Envelope')
    plt.title(f"Sifting Process Iteration {i+1}")
    plt.xlabel("Time [s]")
    plt.ylabel("Amplitude")
    plt.legend()
    
    plt.subplot(2, 1, 2)
    plt.plot(t, imf, 'purple', label='IMF')
    plt.title(f"Intrinsic Mode Function {i+1}")
    plt.xlabel("Time [s]")
    plt.ylabel("Amplitude")
    plt.legend()
    
    plt.tight_layout()
    plt.show()
    
    # 남은 신호 업데이트
    residual -= imf
