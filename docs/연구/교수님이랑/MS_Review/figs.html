2. 퓨리에변환

spec <- function(y){
    n = length(y)
    spec = (abs(fft(y)))/n
    omega = 1:n/n 
    if(n%%2 ==0){
        m = (n/2)+1
    } else {
        m = (n-1)/2+1
    }
    spec = c(spec[(m+1):n],spec[1:m])
    omega = c(omega[(m+1):n]-1,omega[1:m]) -1/n
    list(freq=omega*2*pi,spec=spec)
}       
t = (1:40)/40
eps = rnorm(40)
const = -2+t*0
low = 9*sin(2*pi*t)
high = 4*cos(4*2*pi*t)
ytrue = const + low + high
y = ytrue + eps
freq = spec(y)$freq 
y_spec = spec(y)$spec 
const_spec = spec(const)$spec
low_spec = spec(low)$spec
high_spec = spec(high)$spec
eps_spec = spec(eps)$spec
figsize(8,6)
