Author

Published

January 18, 2024

- 환경셋팅

conda create -n pyod 
conda activate pyod
conda install -c conda-forge pyod

- 실행해볼코드

from pygod.utils import load_data
from pygod.detector import DOMINANT

data = load_data('inj_cora')
detector = DOMINANT(hid_dim=64, num_layers=4, epoch=100)
detector.fit(data)
