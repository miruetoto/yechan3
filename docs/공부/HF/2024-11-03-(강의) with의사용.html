# 예제1 – 기본예제

- context manager

class Dummy: 
    def __enter__(self):
        print("enter")
    def __exit__(self,*args): # *args는 에러처리 관련된 기본변수들
        print("exit")

- 아래의 2개는 같은코드임..

d = Dummy()
d.__enter__()

d.__exit__()

# 예제2 – 타이머

class Timer:
    def __enter__(self):
        self.start = time.time()
    def __exit__(self,*args):
        self.end = time.time()
        print(f'{self.end- self.start:.4f} 초 걸림')
with Timer():
    np.random.randn(10000,10000) + 30
1.5234 초 걸림

# 예제4 – 크기조정

class FigureSizeContext:
    def __enter__(self):
        def resize(w,h):
            plt.rcParams['figure.figsize'] = (w,h)
        return resize
    def __exit__(self,*args):
        plt.rcdefaults()

- 사용예시1

context_manager = FigureSizeContext()
resize = context_manager.__enter__()
resize(3,3)
plt.plot([1,2,3],'--o')
context_manager.__exit__()
plt.plot([1,2,2],'--o')

- 사용예시2

context_manager = FigureSizeContext()
with context_manager as resize:
    resize(3,3)
    plt.plot([1,2,3],'--o')
plt.plot([1,2,2],'--o')

- 정리하면 아래는 같은코드이다.

### 코드1
context_manager = FigureSizeContext()
resize = context_manager.__enter__()
resize(3,3)
plt.plot([1,3,2],'--o')
context_manager.__exit__()
## 코드2
context_manager = FigureSizeContext()
with context_manager as resize:
    resize(3,3)
    plt.plot([1,3,2],'--o')
## 코드3
with FigureSizeContext() as resize:
    resize(3,3)
    plt.plot([1,3,2],'--o')

# 예제5 – 샌드위치만들기

class 샌드위치생성기:
    def __init__(self, 빵="기본빵"):
        self.빵 = 빵
        self.재료들 = []
    def __enter__(self):
        print(f'---{self.빵}---')    
        return self # 보통이런패턴..
    def __exit__(self, *args):
        print(f'---{self.빵}---')
    def 재료추가하기(self,재료):
        print(재료)
        self.재료들.append(재료)

- 사용예시1

with 샌드위치생성기() as 샌드위치1:
    pass

- 사용예시2

with 샌드위치생성기(빵='허니오토') as 샌드위치2:

    샌드위치2.재료추가하기('토마토')
